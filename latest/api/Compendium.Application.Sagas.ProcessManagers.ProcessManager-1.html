Base class for DDD-style orchestration sagas (Process Managers). Subclass this to declare a saga's steps and state shape; the IProcessManagerOrchestrator drives execution and compensation.

public abstract class ProcessManager<TState> : IProcessManager<TState>, IProcessManager, ISagaInstance, IMutableProcessManager where TState : class, new()

Examples

public sealed class OrderProcessManager : ProcessManager<OrderProcessState>
{
    private OrderProcessManager(Guid id, OrderProcessState state)
        : base(id, state, new[] { "ReserveInventory", "ProcessPayment", "ShipOrder" })
    {
    }

    public static OrderProcessManager Create(string orderId, decimal amount) =>
        new(Guid.NewGuid(), new OrderProcessState { OrderId = orderId, Amount = amount });
}
